122def train_test_assign( trial_ns, fold=4, use_random=True ):
123    num_tr = len(trial_ns)
124    if use_random:
125        permu = np.random.permutation(num_tr)
126        xtr = np.sort( trial_ns[ permu[range(np.floor(num_tr/fold).astype(int))] ]) 
127        utr = np.sort( trial_ns[ permu[range(np.floor(num_tr/fold).astype(int), num_tr)] ])
128    else:
129        xlist = np.arange(fold//2, num_tr, fold, dtype='int32')
130        ulist = np.setdiff1d(np.arange(num_tr), xlist)
131        xtr = trial_ns[xlist]
132        utr = trial_ns[ulist]
133    return utr, xtr

136def drift_design_matrix(num_trials, drift_spacing=100, to_plot=False):
137    xminus = 1-np.array(range(drift_spacing))/drift_spacing
138    xplus = np.array(range(drift_spacing))/drift_spacing

140    anchor_pnts = list(range(0, num_trials, drift_spacing))
141    NW = len(anchor_pnts)-1
142
143    Xoffset = np.zeros([num_trials, NW], dtype='float32')
144    for nn in range(NW):
145        #if nn < (NW-1):
146        Xoffset[range(drift_spacing*nn, drift_spacing*(nn+1)),nn] = xminus
147        if nn > 0:
148            Xoffset[range(drift_spacing*(nn-1), drift_spacing*(nn)),nn] = xplus
149    #Xoffset[win_spacing*(NW-1):,nn] = 1.0
150    if to_plot:
151        plt.imshow(Xoffset, aspect='auto')
152        plt.show()

157def vector_regression(xs, ys):
158    svec = np.divide(
159        np.mean(np.multiply(ys, np.expand_dims(xs, axis=1)), axis=0) - np.mean(xs)*np.mean(ys, axis=0),
160        np.var(xs))
161    bvec = np.mean(ys, axis=0)-svec*np.mean(xs)
162    return svec, bvec

173# Parse model terms to separate and eliminate variance due to drift and generate other variances
174def adjR2( damod, s, r, i_c, i_u ):
175    """Calculate R2s accounting for predicted drift. In poorly predicted cells, most of the variance is due to
176    drift, so this would make it look like the models are good at predicting stim and choice, when in fact they
177    are just predicting drifts."""
178    num_cells = r.shape[1]
179    # Robs adjusted for drift (to extent possible with model)
180    drift = damod.generate_prediction(input_data=s, ffnet_target=0)
181    driftC = drift[i_c,:]
182    driftU = drift[i_u,:]
183    drift_var_frac = np.divide( np.var(drift, axis=0), np.var(r, axis=0))
184    
185    RadjC = deepcopy(r[i_c,:])-driftC
186    RadjU = deepcopy(r[i_u,:])-driftU
187    predCadj = damod.generate_prediction(input_data=s, data_indxs=i_c)-driftC
188    predUadj = damod.generate_prediction(input_data=s, data_indxs=i_u)-driftU
189
190    # Adjusted R2s for drift
191    R2s = np.zeros([num_cells,2])
192    R2s[:,0] = 1 - np.divide( np.mean(np.square(RadjC-predCadj), axis=0), np.var(RadjC, axis=0) )
193    R2s[:,1] = 1 - np.divide( np.mean(np.square(RadjU-predUadj), axis=0), np.var(RadjU, axis=0) )
194    return R2s, drift_var_frac

236def mycorr( a, b ):
237    anorm = (deepcopy(a)-np.mean(a)) / np.maximum(np.std(a), 1e-6)
238    bnorm = (deepcopy(b)-np.mean(b)) / np.maximum(np.std(b), 1e-6)
239    cor = np.mean(np.multiply(anorm, bnorm))
240    return cor

243def calc_pop_angle( v1, v2 ):
244    nv1 = deepcopy(v1) / np.maximum(vec_norm(v1), 1e-6)
245    nv2 = deepcopy(v2) / np.maximum(vec_norm(v2), 1e-6)

316def parse_channel_name( cn, area=None ):
317    indices = [i for i, a in enumerate(cn) if a == '_']
318    exptname = cn[:indices[1]]
319    if len(indices) == 5:
320        # then only one electrode in experiment
321        chname = cn[indices[1]+1:indices[3]]
322        head = None
323    else:
324        if cn[indices[1]+2] == 'A':
325            head = 'a'
326        elif cn[indices[1]+2] == 'B':
327            head = 'b'
328        elif cn[indices[1]+2] == 'C':
329            head = 'c'
330        elif cn[indices[1]+2] == 'D':
331            head = 'd'
332        else:
333            print( 'problem parsing headstage info:', cn )
334        chname = cn[indices[2]+1:indices[4]]
335
336    if area is None:
337        return exptname, head, chname
338    else:
339        if head is None:
340            fname = exptname+'_'+area+'_py.mat'
341        else:
342            fname = exptname+'_'+area+head+'_py.mat'
343        return fname, chname
