169        """
170        Prepares stimulus for dataset.
171
172        Args:
173            which_stim: 'left' or 'right' (default 'left')
174            skip_lags: how many lags to skip (default 2)
175            num_lags: how many lags to include in stimulus (default None, will use dataset value)
176
177        Returns:
178            None
179        """
180        if skip_lags is not None:  
181            self.skip_lags = skip_lags
182        # otherwise will use already set value
183            
184        if which_stim in ['left', 'L', 'Left']:
185            stim = torch.tensor( self.stimL, dtype=torch.float32 )
186        else:
187            stim = torch.tensor( self.stimR, dtype=torch.float32 )
188
189        # Zero out invalid time points (disp=0) before time embedding
190        df_generic = torch.zeros( stim.shape, dtype=torch.float32 )
191        df_generic[self.used_inds, :] = 1.0
192        stim = stim * df_generic

199            stim[:self.skip_lags, :] = 0.0
200        elif self.skip_lags < 0:
201            print("Currently cannot use negative skip_lags, and doesnt make sense anyway")
202            self.skip_lags = 0
203
204        if num_lags is None:
205            # then read from dataset (already set):
206            num_lags = self.num_lags

249    def autoencoder_design_matrix( self, pre_win=0, post_win=0, blank=0, cells=None ):
250        """
251        Makes auto-encoder input using windows described above, and including the
252        chosen cells. Will put as additional covariate "ACinput" in __get_item__
253        
254        Args:
255            pre_win: how many time steps to include before origin
256            post_win: how many time steps to include after origin
257            blank: how many time steps to blank in each direction, including origin
258        
259        Returns:
260            None
261        """
262
263        if cells is None:
264            cells = np.arange(self.NC)
265        Rraw = deepcopy(self.robs[:, cells])
266        self.ACinput = torch.zeros(Rraw.shape, dtype=torch.float32)
267        nsteps = 0
268        if blank == 0:
269            self.ACinput += Rraw
270            nsteps = 1
271        for ii in range(1, (pre_win+1)):
272            self.ACinput[ii:, :] += Rraw[:(-ii), :]
273            nsteps += 1
274        for ii in range(1, (post_win+1)):
275            self.ACinput[:(-ii), :] += Rraw[ii:, :]
276            nsteps += 1
277        assert nsteps > 0, "autoencoder design: invalid parameters"
278        self.ACinput *= 1.0/nsteps
279        self.ACinput *= self.dfs[:, cells]

311    def train_test_assign( trial_ns, fold=4, use_random=True ):  # this should be a static function
312        """
313        Assigns trials to training and test sets.
314
315        Args:
316            trial_ns: trial numbers
317            fold: number of folds (default 4)
318            use_random: whether to use random assignment (default True)
319
320        Returns:
321            utr: trial numbers for training set
322            xtr: trial numbers for test set
323        """
324        num_tr = len(trial_ns)
325        if use_random:
326            permu = np.random.permutation(num_tr)
327            xtr = np.sort( trial_ns[ permu[range(np.floor(num_tr/fold).astype(int))] ]) 
328            utr = np.sort( trial_ns[ permu[range(np.floor(num_tr/fold).astype(int), num_tr)] ])
329        else:
330            xlist = np.arange(fold//2, num_tr, fold, dtype='int32')
331            ulist = np.setdiff1d(np.arange(num_tr), xlist)
332            xtr = trial_ns[xlist]
333            utr = trial_ns[ulist]
334        return utr, xtr
