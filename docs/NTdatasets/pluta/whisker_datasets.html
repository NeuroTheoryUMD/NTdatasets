330    def temporal_basis( self, nlags, anchors ):
331        """
332        Make temporal basis for use in temporal basis expansion.
333
334        Args:
335            nlags: number of lags
336            anchors: list of anchor points for the temporal basis
337
338        Returns:
339            KB: temporal basis
340        """
341        if anchors[-1] < nlags:
342            anchors.append(nlags)
343        anchors = np.array(anchors, dtype=np.int64)-1
344        num_tk=len(anchors)-1
345        KB = np.zeros([nlags, num_tk])
346        KB[anchors[0]+1,0]=1
347        for ii in range(1,num_tk):
348            if anchors[ii]-anchors[ii-1] > 1:  # then need to slope
349                dx = anchors[ii]-anchors[ii-1]
350                KB[np.arange(anchors[ii-1]+1, anchors[ii]+1)+1, ii] = (np.arange(dx)+1)/dx
351            else:
352                KB[anchors[ii]+1,ii] = 1 
353            if anchors[ii+1]-anchors[ii] > 1:  # then need to slope
354                dx = anchors[ii+1]-anchors[ii]
355                KB[np.arange(anchors[ii], anchors[ii+1])+1, ii] = 1-(np.arange(dx))/dx
356        print(num_tk, 'basis elements')
357        return KB

359    def anchor_set( self, nlags, doubling_time, offset=0 ):
360        """
361        Make anchor set for temporal basis expansion.
362
363        Args:
364            nlags: number of lags
365            doubling_time: doubling time for temporal basis
366            offset: offset for anchor set
367
368        Returns:
369            a: anchor set
370        """
371        sp = 1
372        a = []
373        x = offset
374        while x < nlags:
375            for ii in range(doubling_time):
376                x += sp
377                if x < nlags:
378                    a.append(x)
379            sp *= 2
380        if a[-1] >= nlags-2:
381            a[-1] = nlags
382        else:
383            a.append(nlags)
384        return a

333        Make temporal basis for use in temporal basis expansion.
334
335        Args:
336            nlags: number of lags
337            anchors: list of anchor points for the temporal basis
338
339        Returns:
340            KB: temporal basis
341        """
342        if anchors[-1] < nlags:
343            anchors.append(nlags)
344        anchors = np.array(anchors, dtype=np.int64)-1
345        num_tk=len(anchors)-1
346        KB = np.zeros([nlags, num_tk])
347        KB[anchors[0]+1,0]=1
348        for ii in range(1,num_tk):
349            if anchors[ii]-anchors[ii-1] > 1:  # then need to slope
350                dx = anchors[ii]-anchors[ii-1]
351                KB[np.arange(anchors[ii-1]+1, anchors[ii]+1)+1, ii] = (np.arange(dx)+1)/dx
352            else:
353                KB[anchors[ii]+1,ii] = 1 
354            if anchors[ii+1]-anchors[ii] > 1:  # then need to slope
355                dx = anchors[ii+1]-anchors[ii]
356                KB[np.arange(anchors[ii], anchors[ii+1])+1, ii] = 1-(np.arange(dx))/dx
357        print(num_tk, 'basis elements')

361        """
362        Make anchor set for temporal basis expansion.
363
364        Args:
365            nlags: number of lags
366            doubling_time: doubling time for temporal basis
367            offset: offset for anchor set
368
369        Returns:
370            a: anchor set
371        """
372        sp = 1
373        a = []
374        x = offset
375        while x < nlags:
376            for ii in range(doubling_time):
377                x += sp
378                if x < nlags:
379                    a.append(x)
380            sp *= 2
381        if a[-1] >= nlags-2:
382            a[-1] = nlags
383        else:
384            a.append(nlags)
385        return a
